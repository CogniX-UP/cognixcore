    def write(*args):
        if not InfoMsgs.enabled:
            return

        s = ''
        for arg in args:
            s += ' '+str(arg)
        print('--> INFO: '+s)

    def write_err(*args):
        if not (InfoMsgs.enabled or InfoMsgs.enabled_errors):
            return

        s = ''
        for arg in args:
            s += ' '+str(arg)

        sys.stderr.write(s)

        if InfoMsgs.traceback_enabled:
            sys.stderr.write(traceback.format_exc())
